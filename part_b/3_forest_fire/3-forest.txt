import csv

    month = row[2]
    temp = float(row[8])
    print(f"{month}\t{temp}")

#CELL 4
%%writefile reducer.py
import sys

current_month = None
temp_sum = 0
count = 0

for line in sys.stdin:
    month, temp = line.strip().split("\t")
    temp = float(temp)

    if current_month == month:
        temp_sum += temp
        count += 1
    else:
        if current_month:
            print(f"{current_month}\t{temp_sum / count}")
        current_month = month
        temp_sum = temp
        count = 1

if current_month:
    print(f"{current_month}\t{temp_sum / count}")

#CELL 13
import findspark
findspark.init()

from pyspark.sql import SparkSession

spark = SparkSession.builder.master("local[*]").getOrCreate()

#CELL 14
df = spark.read.csv("forestfires.csv", header=True, inferSchema=True)
df.show(5)

#CELL 15
df.createOrReplaceTempView("forestfires")

#CELL 16
spark.sql("""
SELECT month, AVG(temp) as avg_temp
FROM forestfires
GROUP BY month
""").show()

#CELL 17
spark.sql("""
SELECT wind, AVG(area) as avg_area
FROM forestfires
GROUP BY wind
ORDER BY wind
""").show()

#CELL 18
spark.sql("""
SELECT * FROM forestfires
WHERE temp > 30 AND wind > 5
""").show()
